  <script>

    function clearErrors() {

      for (let i = 1; i <= 5; i++) {

        const errEl = document.getElementById("error-q" + i);

        if (errEl) errEl.style.display = "none";

      }

    }


    function calculateScore() {

      clearErrors();


      const form = document.getElementById("iiefForm");

      let total = 0;

      let valid = true;


      for (let i = 1; i <= 5; i++) {

        const selected = form.querySelector('input[name="q' + i + '"]:checked');

        if (!selected) {

          valid = false;

          const errEl = document.getElementById("error-q" + i);

          if (errEl) errEl.style.display = "block";

        } else {

          total += parseInt(selected.value, 10);

        }

      }


      const resultBox = document.getElementById("result");

      const scoreEl = document.getElementById("result-score");

      const categoryEl = document.getElementById("result-category");

      const noteEl = document.getElementById("result-note");


      if (!valid) {

        resultBox.style.display = "none";

        return;

      }


      // Determine category

      let categoryText = "";

      let categoryClass = "";

      let noteText =

        "建議與專業醫療人員討論結果，以獲得進一步評估與建議。";


      if (total >= 22 && total <= 25) {

        categoryText = "無勃起功能障礙";

        categoryClass = "severity-none";

      } else if (total >= 17 && total <= 21) {

        categoryText = "輕度勃起功能障礙";

        categoryClass = "severity-mild";

      } else if (total >= 12 && total <= 16) {

        categoryText = "輕度至中度勃起功能障礙";

        categoryClass = "severity-mild-moderate";

      } else if (total >= 8 && total <= 11) {

        categoryText = "中度勃起功能障礙";

        categoryClass = "severity-moderate";

      } else if (total >= 5 && total <= 7) {

        categoryText = "重度勃起功能障礙";

        categoryClass = "severity-severe";

      } else {

        categoryText = "分數超出一般量表範圍，請確認作答。";

        noteText = "";

      }


      scoreEl.textContent = "您的總分為：" + total + " 分（理論範圍約 5–25 分）";


      categoryEl.textContent = "初步評估結果：" + categoryText;

      categoryEl.className = "result-line " + categoryClass;


      noteEl.textContent = noteText;


      resultBox.style.display = "block";

    }


    function resetForm() {

      const form = document.getElementById("iiefForm");

      form.reset();

      clearErrors();

      const resultBox = document.getElementById("result");

      resultBox.style.display = "none";

    }
